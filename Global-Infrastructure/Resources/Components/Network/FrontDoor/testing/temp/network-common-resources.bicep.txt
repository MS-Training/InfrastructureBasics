// Create Public IP Prefix
module createPublicIpPrefix 'Network-template-network-publicIpPrefix.bicep' = {
  name: 'CreatePublicIpPrefix-${date}'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  params: {
    publicIpPrefixName: toLower('${resourceNames.publicIpPrefixBaseName}${settings.common.tag}')
    ipTagName: contains(toUpper(resourceNames.environment), 'PRD') ? 'APInvoiceServices' : 'APInvoiceServicesTest'
    location: settings.common.region
  }
}

// Create NAT Gateway
module createNatGateway 'Network-template-network-natGateway.bicep' = {
  name: 'CreateNATGateway-${date}'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  dependsOn: [
    createPublicIpPrefix
  ]
  params: {
    natGatewayName: natGatewayName
    location: settings.common.region
    natGatewayProperties: {
      idleTimeoutInMinutes: '4'
      publicIpPrefixes: [
        {
          id: createPublicIpPrefix.outputs.publicIpPrefixResourceId
        }
      ]
    }
  }
}

// Create Main Virtual Network (only on first deployment)
module createMainVirtualNetwork 'Network-template-network-vnet.bicep' = if (settings.commonNetwork.isFirstDeployment) {
  name: 'CreateMainVirtualNetwork-${date}'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  params: {
    virtualNetworkName: virtualNetworkName
    virtualNetworkAddressPrefix: settings.commonNetwork.virtualNetworkAddressPrefix
    location: settings.common.region
  }
}

// Create Subnets NSGs (only on first deployment)
module createSubnetsNsgs 'Network-template-network-nsg.bicep' = [for (subnet, i) in settings.commonNetwork.subnets: if (settings.commonNetwork.isFirstDeployment) {
  name: 'createSubnetsNsgs-${subnet.subnetName}'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  params: {
    subnetName: subnet.subnetName
  }
}]

// Create Subnets
module createSubnets 'Network-template-network-subnet.bicep' = [for (subnet, i) in settings.commonNetwork.subnets: {
  name: 'createSubnet-${subnet.subnetName}'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  dependsOn: [
    createMainVirtualNetwork
    createNatGateway
    createSubnetsNsgs
  ]
  params: {
    virtualNetworkName: virtualNetworkName
    subnetName: subnet.subnetName
    natGatewayResourceId: createNatGateway.outputs.natGatewayResourceId
    subnetAddressPrefix: subnet.subnetAddressPrefix
    delegations: subnet.delegations
    serviceEndpoints: subnet.serviceEndpoints
  }
}]

// Create Subnets NSG Rules
module createSubnetsNsgRules 'Network-template-network-nsg-rules.bicep' = [for (subnet, i) in settings.commonNetwork.subnets: {
  name: 'createSubnetsRules-${subnet.subnetName}'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  dependsOn: [
    createSubnetsNsgs
    createSubnets
  ]
  params: {
    subnetName: subnet.subnetName
    ruleDetailsList: standardNsgRules
  }
}]

// Create Private DNS Zones
module createPrivateDNSZones 'Network-template-network-privateDNSZone.bicep' = [for (zone, i) in privateDNSZoneList: {
  name: 'createPrivateDNSZones-${take(zone, 65)}'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  dependsOn: [
    createMainVirtualNetwork
  ]
  params: {
    privateDNSZoneName: zone
    virtualNetworkResourceId: resourceNames.commonVirtualNetworkResourceId
  }
}]

// Create Azure Front Door WAF Policy (only in primary region)
module createAzureFrontDoorWAFPolicy 'Network-template-network-webApplicationFirewall.bicep' = if (settings.common.isPrimaryRegion) {
  name: 'CreateAzureFrontDoorWAFPolicy-${date}'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  params: {
    wafPolicyName: frontDoorWafPolicyName
    rateLimitThreshold: settings.commonNetwork.frontDoorWafConfig.rateLimitThreshold
  }
}

// Create Azure Front Door (only in primary region)
module createAzureFrontDoor 'CdnProfiles-template-cdn-profile.bicep' = if (settings.common.isPrimaryRegion) {
  name: 'CreateAzureFrontDoor-${date}'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  params: {
    cdnProfileName: cdnProfileName
    skuName: settings.commonNetwork.frontDoorConfig.skuName
    originResponseTimeoutSeconds: 60
    workSpaceResourceId: resourceNames.commonWorkSpaceResourceId
  }
}

// Create Front Door Endpoints (only in primary region)
module createFrontDoorEndPoints 'CdnProfiles-template-cdn-profile-afdEndPoint.bicep' = [for (ep, i) in settings.commonNetwork.frontDoorConfig.endPointsConfiguration: if (settings.common.isPrimaryRegion) {
  name: 'Create-${ep.mainServiceName}-${ep.serviceRoleName}-FrontDoorEndPoint'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  dependsOn: [
    createAzureFrontDoor
  ]
  params: {
    cdnProfileName: cdnProfileName
    afdEndPointName: toLower('${ep.mainServiceName}${ep.serviceRoleName}${resourceNames.environment}')
  }
}]

// Create Front Door Custom Domains (only in primary region)
module createFrontDoorCustomDomains 'CdnProfiles-template-cdn-profile-customDomain.bicep' = [for (ep, i) in settings.commonNetwork.frontDoorConfig.endPointsConfiguration: if (settings.common.isPrimaryRegion) {
  name: 'Create-${ep.mainServiceName}-${ep.serviceRoleName}-FrontDoorCustomDomain'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  dependsOn: [
    createAzureFrontDoor
  ]
  params: {
    cdnProfileName: cdnProfileName
    customDomainHostName: toLower('${ep.mainServiceName}.${ep.serviceRoleName}.${resourceNames.environment}.${ep.domainName}')
  }
}]

// Create Front Door Security Policy WAF (only in primary region)
module createFrontDoorSecurityPolicyWAF 'CdnProfiles-template-cdn-profile-securityPolicies.bicep' = [for (ep, i) in settings.commonNetwork.frontDoorConfig.endPointsConfiguration: if (settings.common.isPrimaryRegion) {
  name: 'Create-${ep.mainServiceName}-${ep.serviceRoleName}-FrontDoorCustomDomainWAF'
  scope: resourceGroup(resourceNames.commonNetworkResourceGroupName)
  dependsOn: [
    createAzureFrontDoor
    createAzureFrontDoorWAFPolicy
    createFrontDoorCustomDomains
  ]
  params: {
    profileName: cdnProfileName
    wafPolicyId: createAzureFrontDoorWAFPolicy.outputs.frontDoorWafId
    afdEndpoint: createFrontDoorCustomDomains[i].outputs.afdCustomEndpoint
  }
}]
